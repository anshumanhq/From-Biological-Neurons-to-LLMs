% GPT (2018) – Core Equations

% 1. Autoregressive language modelling objective (pre-training):
\begin{equation}
\mathcal{L}_1(U) = -\sum_{i} \log P(u_i \mid u_{i-k}, \dots, u_{i-1}; \Theta)
\end{equation}

% 2. Transformer decoder layer (self-attention with causal masking):
\begin{equation}
h_l = \text{block}(h_{l-1}) \quad \forall l \in [1, L]
\end{equation}

% 3. Causal masked self-attention:
\begin{equation}
\text{Attention}(Q, K, V) = \text{softmax}\left( \frac{QK^T}{\sqrt{d_k}} + M \right) V
\end{equation}
where \(M\) is a mask matrix (upper triangular of \(-\infty\)) to prevent attending to future tokens.

% 4. Position-wise feed-forward network:
\begin{equation}
\text{FFN}(x) = \max(0, xW_1 + b_1) W_2 + b_2
\end{equation}

% 5. Supervised fine-tuning objective:
\begin{equation}
P(y \mid x^1, \dots, x^m) = \text{softmax}(h_l^m W_y)
\end{equation}

% 6. Combined objective (pre-training + fine-tuning):
\begin{equation}
\mathcal{L}_3 = \mathcal{L}_2 + \lambda \mathcal{L}_1
\end{equation}
