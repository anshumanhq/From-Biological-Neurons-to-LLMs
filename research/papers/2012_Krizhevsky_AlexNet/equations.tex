% AlexNet (2012) – Core Equations

% 1. Convolution operation (standard):
\begin{equation}
\mathbf{S}_{i,j} = \sum_{u=1}^{k} \sum_{v=1}^{k} \mathbf{X}_{i+u-1, j+v-1} \cdot \mathbf{K}_{u,v} + b
\end{equation}

% 2. ReLU activation:
\begin{equation}
f(x) = \max(0, x)
\end{equation}

% 3. Max pooling:
\begin{equation}
\mathbf{P}_{i,j} = \max_{u,v \in \text{window}} \mathbf{S}_{2i+u, 2j+v}
\end{equation}

% 4. Dropout (applied during training):
\begin{equation}
\mathbf{h}_{\text{drop}} = \mathbf{h} \odot \mathbf{m}, \quad m_i \sim \text{Bernoulli}(p)
\end{equation}

% 5. Cross-entropy loss (for classification):
\begin{equation}
\mathcal{L} = -\sum_{c=1}^{C} y_c \log \hat{y}_c
\end{equation}
