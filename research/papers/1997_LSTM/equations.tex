% LSTM (1997) – Core Equations

% 1. Forget gate:
\begin{equation}
f_t = \sigma\left( W_f \cdot [h_{t-1}, x_t] + b_f \right)
\end{equation}

% 2. Input gate:
\begin{equation}
i_t = \sigma\left( W_i \cdot [h_{t-1}, x_t] + b_i \right)
\end{equation}

% 3. Candidate cell state:
\begin{equation}
\tilde{c}_t = \tanh\left( W_c \cdot [h_{t-1}, x_t] + b_c \right)
\end{equation}

% 4. Cell state update:
\begin{equation}
c_t = f_t \odot c_{t-1} + i_t \odot \tilde{c}_t
\end{equation}

% 5. Output gate:
\begin{equation}
o_t = \sigma\left( W_o \cdot [h_{t-1}, x_t] + b_o \right)
\end{equation}

% 6. Hidden state (output) update:
\begin{equation}
h_t = o_t \odot \tanh(c_t)
\end{equation}

% 7. Constant Error Carousel (CEC) – linear recurrence allowing gradient flow:
\begin{equation}
\frac{\partial c_t}{\partial c_{t-1}} = f_t \quad \text{(nearly 1 when forget gate is open)}
\end{equation}
