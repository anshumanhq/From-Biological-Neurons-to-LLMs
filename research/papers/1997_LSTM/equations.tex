% Original 1997 LSTM (without forget gate)

% 1. Input gate:
\begin{equation}
i_t = \sigma\left( W_i \cdot [h_{t-1}, x_t] + b_i \right)
\end{equation}

% 2. Candidate cell state:
\begin{equation}
\tilde{c}_t = \tanh\left( W_c \cdot [h_{t-1}, x_t] + b_c \right)
\end{equation}

% 3. Cell state update (Constant Error Carousel – linear with weight 1):
\begin{equation}
c_t = c_{t-1} + i_t \odot \tilde{c}_t
\end{equation}

% 4. Output gate:
\begin{equation}
o_t = \sigma\left( W_o \cdot [h_{t-1}, x_t] + b_o \right)
\end{equation}

% 5. Hidden state (output) update:
\begin{equation}
h_t = o_t \odot \tanh(c_t)
\end{equation}

% 6. Constant Error Carousel – gradient flow (idealised):
\begin{equation}
\frac{\partial c_t}{\partial c_{t-1}} = 1 \quad \text{(when no external interference)}
\end{equation}
