% Hebb (1949) – Core Mathematical Abstraction
% 
% Although Hebb did not write this equation explicitly, it is universally recognized 
% as the formalization of his neurophysiological postulate.

\begin{equation}
\boxed{\Delta w_{ij} = \eta \cdot a_i \cdot a_j}
\end{equation}

Where:
\begin{itemize}
    \item $\Delta w_{ij}$ = Change in synaptic efficacy from neuron $i$ to neuron $j$
    \item $\eta$ = Learning rate (positive constant)
    \item $a_i$ = Activity of presynaptic neuron $i$
    \item $a_j$ = Activity of postsynaptic neuron $j$
\end{itemize}
