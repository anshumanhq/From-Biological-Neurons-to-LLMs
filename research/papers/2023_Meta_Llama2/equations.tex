% Llama 2 (2023) – Core Equations

% 1. Autoregressive language modelling:
\begin{equation}
\mathcal{L} = -\sum_{i} \log P(u_i \mid u_{<i}; \Theta)
\end{equation}

% 2. Causal masked self-attention:
\begin{equation}
\text{Attention}(Q, K, V) = \text{softmax}\left( \frac{QK^T}{\sqrt{d_k}} + M \right) V
\end{equation}

% 3. RMSNorm (as in LLaMA):
\begin{equation}
\text{RMSNorm}(x) = \frac{x}{\sqrt{\frac{1}{n} \sum_i x_i^2 + \epsilon}} \odot g
\end{equation}

% 4. SwiGLU activation:
\begin{equation}
\text{SwiGLU}(x) = \left(\text{Swish}(xW_g) \odot xW_u\right)W_d
\end{equation}

% 5. RLHF / PPO (similar to InstructGPT):
\begin{equation}
\mathcal{L}_{\text{RL}} = -\mathbb{E}_{x \sim \mathcal{D}_{\text{RL}}, y \sim \pi_{\theta}(x)}
\left[ r_{\phi}(x, y) - \beta \cdot \text{KL}\left( \pi_{\theta} \parallel \pi_{\text{ref}} \right) \right]
\end{equation}
